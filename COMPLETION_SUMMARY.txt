## 🔧 Usage Example

```python
import asyncio
from jarvis_os import JarvisOS, AgentConfig

async def main():
    # Create agent
    config = AgentConfig(name="my-agent")
    agent = JarvisOS(config)
    await agent.start()
    
    # Execute task
    result = await agent.execute_task(
        task_type="work",
        task_func=lambda x: x * 2,
        task_params={"args": [21]},
    )
    
    print(f"Result: {result.result}")  # Output: 42
    
    # Get metrics
    metrics = agent.get_metrics()
    print(f"Success rate: {metrics['performance']['success_rate']:.1%}")
    
    await agent.stop()

asyncio.run(main())
```
